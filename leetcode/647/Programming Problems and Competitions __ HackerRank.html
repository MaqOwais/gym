    int C = 0, R = -1, rad; 
for (int i = 0; i < T.length(); ++i) {
    if (i <= R) {
        rad = min(P[2*C-i], R-i); 
    } else {
        rad = 0; 
    }
    // Try to extend
    while (i+rad < T.length() && i-rad >= 0 && T[i-rad] == T[i+rad]) {
        rad++; 
    }
    P[i] = rad;
    if (i + rad - 1 > R) {
        C = i; 
        R = i + rad - 1; 
    }
}
